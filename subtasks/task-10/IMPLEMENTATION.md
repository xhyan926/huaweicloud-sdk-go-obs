// examples/post_upload_sample.go

package main

import (
    "fmt"
    "log"
    "os"
)

import (
    obs "github.com/huaweicloud/huaweicloud-sdk-go-obs/obs"
)

func main() {
    // 1. 创建 OBS 客户端
    ak := os.Getenv("OBS_AK")
    sk := os.Getenv("OBS_SK")
    endpoint := os.Getenv("OBS_ENDPOINT")

    if ak == "" || sk == "" || endpoint == "" {
        log.Fatal("Please set OBS_AK, OBS_SK, and OBS_ENDPOINT environment variables")
    }

    obsClient, err := obs.New(ak, sk, endpoint)
    if err != nil {
        log.Fatalf("Failed to create OBS client: %v", err)
    }

    // 2. 创建 POST 上传策略
    bucketName := "your-bucket-name"
    objectKey := "uploads/test.jpg"
    expiresIn := int64(3600) // 1 小时后过期

    input := &obs.CreatePostPolicyInput{
        Bucket: bucketName,
        Key:     objectKey,
        Expires: expiresIn,
        Conditions: []obs.PostPolicyCondition{
            // 添加 content-length-range 限制
            obs.CreateContentLengthRangeCondition(0, 10*1024*1024), // 最大 10MB
            // 可以添加更多条件
        },
    }

    output, err := obsClient.CreatePostPolicy(input)
    if err != nil {
        log.Fatalf("Failed to create POST policy: %v", err)
    }

    // 3. 输出策略信息
    fmt.Println("=== POST Upload Policy Generated ===")
    fmt.Printf("Access Key ID: %s\n", output.AccessKeyId)
    fmt.Printf("Policy (Base64): %s\n", output.Policy)
    fmt.Printf("Signature: %s\n", output.Signature)
    fmt.Printf("Complete Token: %s\n\n", output.Token)

    // 4. 生成前端 HTML 表单示例
    fmt.Println("=== Frontend HTML Form Example ===")
    fmt.Println(generateHTMLForm(bucketName, objectKey, output.AccessKeyId, output.Signature, output.Policy))
}

## 技术要点

### POST 上传优势
- 文件直接从浏览器上传到 OBS
- 不经过后端服务器中转
- 降低服务器负载和网络带宽
- 支持大文件上传

### 安全性
- Policy 和签名由后端生成
- 前端只使用生成的凭证
- 可以设置文件大小限制
- 可以设置文件类型限制

### 前端表单要求
- 必须包含 AK（AWSAccessKeyId）
- 必须包含 Policy
- 必须包含 Signature
- 必须包含 Key（对象名）
- 必须包含文件选择器

### 环境变量
- OBS_AK: Access Key ID
- OBS_SK: Secret Access Key
- OBS_ENDPOINT: OBS 端点
